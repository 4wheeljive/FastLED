20i16 float_to_fixed(float f) {
21 f = fl::clamp(f, -1.0f, 1.0f);
22 if (f < 0.0f) {
23 return (i16)(f * INT16_NEG);
24 } else {
25 return (i16)(f * INT16_POS); // Round to nearest
26 }
27}

29// Convert fixed Q15 to float.

32 if (f < 0) {

34 } else {
35 return ((float)f) / INT16_POS; // Positive values
36 }
37}

108void WaveSimulation1D_Real::update() {
109 i16 *curr = (whichGrid == 0) ? grid1.data() : grid2.data();
110 i16 *next = (whichGrid == 0) ? grid2.data() : grid1.data();
111
112 // Update boundaries with a Neumann (zero-gradient) condition:
113 curr[0] = curr[1];
114 curr[length + 1] = curr[length];
115
116 // Compute dampening factor as an integer value: 2^(mDampenening)
117 i32 dampening_factor = 1 << mDampenening;
118
119 i32 mCourantSq32 = static_cast<i32>(mCourantSq);
120 // Iterate over each inner cell.
121 for (fl::size i = 1; i < length + 1; i++) {
122 // Compute the 1D Laplacian:
123 // lap = curr[i+1] - 2 * curr[i] + curr[i-1]
124 i32 lap =
125 (i32)curr[i + 1] - ((i32)curr[i] << 1) + curr[i - 1];
126
127 // Multiply the Laplacian by the simulation speed using Q15 arithmetic:
128 i32 term = (mCourantSq32 * lap) >> 15;
129
130 // Compute the new value:
131 // f = -next[i] + 2 * curr[i] + term
132 i32 f = -(i32)next[i] + ((i32)curr[i] << 1) + term;
133
134 // Apply damping:
135 f = f - (f / dampening_factor);
136
137 // Clamp the result to the Q15 range [-32768, 32767].
138 if (f > 32767)
139 f = 32767;
140 else if (f < -32768)
141 f = -32768;
142
143 next[i] = (i16)f;
144 }
145
146 if (mHalfDuplex) {
147 // Set the negative values to zero.
148 for (fl::size i = 1; i < length + 1; i++) {
149 if (next[i] < 0) {
150 next[i] = 0;
151 }
152 }
153 }
154
155 // Toggle the active grid.
156 whichGrid ^= 1;
157}

226void WaveSimulation2D_Real::update() {
227 i16 *curr = (whichGrid == 0 ? grid1.data() : grid2.data());
228 i16 *next = (whichGrid == 0 ? grid2.data() : grid1.data());
229
230 // Update horizontal boundaries.
231 for (fl::size j = 0; j < height + 2; ++j) {
232 if (mXCylindrical) {
233 curr[j * stride + 0] = curr[j * stride + width];
234 curr[j * stride + (width + 1)] = curr[j * stride + 1];
235 } else {
236 curr[j * stride + 0] = curr[j * stride + 1];
237 curr[j * stride + (width + 1)] = curr[j * stride + width];
238 }
239 }
240
241 // Update vertical boundaries.
242 for (fl::size i = 0; i < width + 2; ++i) {
243 curr[0 * stride + i] = curr[1 * stride + i];
244 curr[(height + 1) * stride + i] = curr[height * stride + i];
245 }
246
247 // Compute the dampening factor as an integer: 2^(dampening).
248 i32 dampening_factor = 1 << mDampening; // e.g., 6 -> 64
249 i32 mCourantSq32 = static_cast<i32>(mCourantSq);
250
251 // Update each inner cell.
252 for (fl::size j = 1; j <= height; ++j) {
253 for (fl::size i = 1; i <= width; ++i) {
254 int index = j * stride + i;
255 // Laplacian: sum of four neighbors minus 4 times the center.
256 i32 laplacian = (i32)curr[index + 1] + curr[index - 1] +
257 curr[index + stride] + curr[index - stride] -
258 ((i32)curr[index] << 2);
259 // Compute the new value:
260 // f = - next[index] + 2 * curr[index] + mCourantSq * laplacian
261 // The multiplication is in Q15, so we shift right by 15.
262 i32 term = (mCourantSq32 * laplacian) >> 15;
263 i32 f =
264 -(i32)next[index] + ((i32)curr[index] << 1) + term;
265
266 // Apply damping:
267 f = f - (f / dampening_factor);
268
269 // Clamp f to the Q15 range.
270 if (f > 32767)
271 f = 32767;
272 else if (f < -32768)
273 f = -32768;
274
275 next[index] = (i16)f;
276 }
277 }
278
279 if (mHalfDuplex) {
280 // Set negative values to zero.
281 for (fl::size j = 1; j <= height; ++j) {
282 for (fl::size i = 1; i <= width; ++i) {
283 int index = j * stride + i;
284 if (next[index] < 0) {
285 next[index] = 0;
286 }
287 }
288 }
289 }
290
291 // Swap the roles of the grids.
292 whichGrid ^= 1;
293}
