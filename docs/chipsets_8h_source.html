219template <
220 uint8_t DATA_PIN, uint8_t CLOCK_PIN,
221 EOrder RGB_ORDER = RGB,
222 uint32_t SPI_SPEED = DATA_RATE_MHZ(12),
223 FiveBitGammaCorrectionMode GAMMA_CORRECTION_MODE = kFiveBitGammaCorrectionMode_Null,
224 uint32_t START_FRAME = 0x00000000,
225 uint32_t END_FRAME = 0xFF000000
226>
227class APA102Controller : public CPixelLEDController<RGB_ORDER> {
228 typedef SPIOutput<DATA_PIN, CLOCK_PIN, SPI_SPEED> SPI;
229 SPI mSPI;
230
231 void startBoundary() {
232 mSPI.writeWord(START_FRAME >> 16);
233 mSPI.writeWord(START_FRAME & 0xFFFF);
234 }
235 void endBoundary(int nLeds) {
236 int nDWords = (nLeds/32);
237 const uint8_t b0 = uint8_t(END_FRAME >> 24 & 0x000000ff);
238 const uint8_t b1 = uint8_t(END_FRAME >> 16 & 0x000000ff);
239 const uint8_t b2 = uint8_t(END_FRAME >> 8 & 0x000000ff);
240 const uint8_t b3 = uint8_t(END_FRAME >> 0 & 0x000000ff);
241 do {
242 mSPI.writeByte(b0);
243 mSPI.writeByte(b1);
244 mSPI.writeByte(b2);
245 mSPI.writeByte(b3);
246 } while(nDWords--);
247 }
248
249 inline void writeLed(uint8_t brightness, uint8_t b0, uint8_t b1, uint8_t b2) __attribute__((always_inline)) {
250#ifdef FASTLED_SPI_BYTE_ONLY
251 mSPI.writeByte(0xE0 | brightness);
252 mSPI.writeByte(b0);
253 mSPI.writeByte(b1);
254 mSPI.writeByte(b2);
255#else
256 uint16_t b = 0xE000 | (brightness << 8) | (uint16_t)b0;
257 mSPI.writeWord(b);
258 uint16_t w = b1 << 8;
259 w |= b2;
260 mSPI.writeWord(w);
261#endif
262 }
263
264 inline void write2Bytes(uint8_t b1, uint8_t b2) __attribute__((always_inline)) {
265#ifdef FASTLED_SPI_BYTE_ONLY
266 mSPI.writeByte(b1);
267 mSPI.writeByte(b2);
268#else
269 mSPI.writeWord(uint16_t(b1) << 8 | b2);
270#endif
271 }
272
273public:
274 APA102Controller() {}
275
276 virtual void init() {
277 mSPI.init();
278 }
279
280protected:
282 virtual void showPixels(PixelController<RGB_ORDER> & pixels) {
283 switch (GAMMA_CORRECTION_MODE) {
284 case kFiveBitGammaCorrectionMode_Null: {
285 showPixelsDefault(pixels);
286 break;
287 }
288 case kFiveBitGammaCorrectionMode_BitShift: {
289 showPixelsGammaBitShift(pixels);
290 break;
291 }
292 }
293 }

297 static inline void getGlobalBrightnessAndScalingFactors(
298 PixelController<RGB_ORDER>& pixels,
299 uint8_t* out_s0, uint8_t* out_s1, uint8_t* out_s2, uint8_t* out_brightness) {
300 uint8_t s0 = pixels.getScale0();
301 uint8_t s1 = pixels.getScale1();
302 uint8_t s2 = pixels.getScale2();
303#if FASTLED_USE_GLOBAL_BRIGHTNESS == 1
304 const uint16_t maxBrightness = 0x1F;
305 uint16_t brightness = ((((uint16_t)max(max(s0, s1), s2) + 1) * maxBrightness - 1) >> 8) + 1;
306 s0 = (maxBrightness * s0 + (brightness >> 1)) / brightness;
307 s1 = (maxBrightness * s1 + (brightness >> 1)) / brightness;
308 s2 = (maxBrightness * s2 + (brightness >> 1)) / brightness;
309#else
310 const uint8_t brightness = 0x1F;
311#endif
312 *out_s0 = s0;
313 *out_s1 = s1;
314 *out_s2 = s2;
315 *out_brightness = static_cast<uint8_t>(brightness);
316 }

318 // Legacy showPixels implementation.
319 inline void showPixelsDefault(PixelController<RGB_ORDER> & pixels) {
320 mSPI.select();
321 uint8_t s0, s1, s2, global_brightness;
322 getGlobalBrightnessAndScalingFactors(pixels, &s0, &s1, &s2, &global_brightness);
323 startBoundary();
324 while (pixels.has(1)) {
325 uint8_t r = pixels.loadAndScale0(0, s0);
326 uint8_t g = pixels.loadAndScale1(0, s1);
327 uint8_t b = pixels.loadAndScale2(0, s2);
328 writeLed(global_brightness, r, g, b);
329 pixels.stepDithering();
330 pixels.advanceData();
331 }
332 endBoundary(pixels.size());
333
334 mSPI.waitFully();
335 mSPI.release();
336 }

338 inline void showPixelsGammaBitShift(PixelController<RGB_ORDER> & pixels) {
339 mSPI.select();
340 uint8_t r_scale = pixels.getScale0();
341 uint8_t g_scale = pixels.getScale1();
342 uint8_t b_scale = pixels.getScale2();
343 startBoundary();
344 while (pixels.has(1)) {
345 // Load raw uncorrected r,g,b values.
346 uint8_t r = pixels.loadAndScale0(0, 0xFF);
347 uint8_t g = pixels.loadAndScale1(0, 0xFF);
348 uint8_t b = pixels.loadAndScale2(0, 0xFF);
349 uint8_t brightness = 0;
350 if ((r | g | b) != 0) {
351 // Compute the gamma correction for non black pixels.
352 five_bit_hd_gamma_bitshift(
353 r, g, b,
354 r_scale, g_scale, b_scale, // Post-gamma scale.
355 &r, &g, &b, &brightness);
356 }
357 writeLed(brightness, r, g, b);
358 pixels.stepDithering();
359 pixels.advanceData();
360 }
361 endBoundary(pixels.size());
362 mSPI.waitFully();
363 mSPI.release();
364 }
