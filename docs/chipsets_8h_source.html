220template <
221 uint8_t DATA_PIN, uint8_t CLOCK_PIN,
222 EOrder RGB_ORDER = RGB,
223 // APA102 has a bug where long strip can't handle full speed due to clock degredation.
224 // This only affects long strips, but then again if you have a short strip does 6 mhz actually slow
225 // you down? Probably not. And you can always bump it up for speed. Therefore we are prioritizing
226 // "just works" over "fastest possible" here.
227 // https://www.pjrc.com/why-apa102-leds-have-trouble-at-24-mhz/
228 uint32_t SPI_SPEED = DATA_RATE_MHZ(6),
229 FiveBitGammaCorrectionMode GAMMA_CORRECTION_MODE = kFiveBitGammaCorrectionMode_Null,
230 uint32_t START_FRAME = 0x00000000,
231 uint32_t END_FRAME = 0xFF000000
232>
233class APA102Controller : public CPixelLEDController<RGB_ORDER> {
234 typedef SPIOutput<DATA_PIN, CLOCK_PIN, SPI_SPEED> SPI;
235 SPI mSPI;
236
237 void startBoundary() {
238 mSPI.writeWord(START_FRAME >> 16);
239 mSPI.writeWord(START_FRAME & 0xFFFF);
240 }
241 void endBoundary(int nLeds) {
242 int nDWords = (nLeds/32);
243 const uint8_t b0 = uint8_t(END_FRAME >> 24 & 0x000000ff);
244 const uint8_t b1 = uint8_t(END_FRAME >> 16 & 0x000000ff);
245 const uint8_t b2 = uint8_t(END_FRAME >> 8 & 0x000000ff);
246 const uint8_t b3 = uint8_t(END_FRAME >> 0 & 0x000000ff);
247 do {
248 mSPI.writeByte(b0);
249 mSPI.writeByte(b1);
250 mSPI.writeByte(b2);
251 mSPI.writeByte(b3);
252 } while(nDWords--);
253 }
254
255 inline void writeLed(uint8_t brightness, uint8_t b0, uint8_t b1, uint8_t b2) __attribute__((always_inline)) {
256#ifdef FASTLED_SPI_BYTE_ONLY
257 mSPI.writeByte(0xE0 | brightness);
258 mSPI.writeByte(b0);
259 mSPI.writeByte(b1);
260 mSPI.writeByte(b2);
261#else
262 uint16_t b = 0xE000 | (brightness << 8) | (uint16_t)b0;
263 mSPI.writeWord(b);
264 uint16_t w = b1 << 8;
265 w |= b2;
266 mSPI.writeWord(w);
267#endif
268 }
269
270 inline void write2Bytes(uint8_t b1, uint8_t b2) __attribute__((always_inline)) {
271#ifdef FASTLED_SPI_BYTE_ONLY
272 mSPI.writeByte(b1);
273 mSPI.writeByte(b2);
274#else
275 mSPI.writeWord(uint16_t(b1) << 8 | b2);
276#endif
277 }
278
279public:
280 APA102Controller() {}
281
282 virtual void init() {
283 mSPI.init();
284 }
285
286protected:
288 virtual void showPixels(PixelController<RGB_ORDER> & pixels) {
289 switch (GAMMA_CORRECTION_MODE) {
290 case kFiveBitGammaCorrectionMode_Null: {
291 showPixelsDefault(pixels);
292 break;
293 }
294 case kFiveBitGammaCorrectionMode_BitShift: {
295 showPixelsGammaBitShift(pixels);
296 break;
297 }
298 }
299 }

303 static inline void getGlobalBrightnessAndScalingFactors(
304 PixelController<RGB_ORDER>& pixels,
305 uint8_t* out_s0, uint8_t* out_s1, uint8_t* out_s2, uint8_t* out_brightness) {
306 uint8_t s0 = pixels.getScale0();
307 uint8_t s1 = pixels.getScale1();
308 uint8_t s2 = pixels.getScale2();
309#if FASTLED_USE_GLOBAL_BRIGHTNESS == 1
310 const uint16_t maxBrightness = 0x1F;
311 uint16_t brightness = ((((uint16_t)max(max(s0, s1), s2) + 1) * maxBrightness - 1) >> 8) + 1;
312 s0 = (maxBrightness * s0 + (brightness >> 1)) / brightness;
313 s1 = (maxBrightness * s1 + (brightness >> 1)) / brightness;
314 s2 = (maxBrightness * s2 + (brightness >> 1)) / brightness;
315#else
316 const uint8_t brightness = 0x1F;
317#endif
318 *out_s0 = s0;
319 *out_s1 = s1;
320 *out_s2 = s2;
321 *out_brightness = static_cast<uint8_t>(brightness);
322 }

324 // Legacy showPixels implementation.
325 inline void showPixelsDefault(PixelController<RGB_ORDER> & pixels) {
326 mSPI.select();
327 uint8_t s0, s1, s2, global_brightness;
328 getGlobalBrightnessAndScalingFactors(pixels, &s0, &s1, &s2, &global_brightness);
329 startBoundary();
330 while (pixels.has(1)) {
331 uint8_t c0 = pixels.loadAndScale0(0, s0);
332 uint8_t c1 = pixels.loadAndScale1(0, s1);
333 uint8_t c2 = pixels.loadAndScale2(0, s2);
334 writeLed(global_brightness, c0, c1, c2);
335 pixels.stepDithering();
336 pixels.advanceData();
337 }
338 endBoundary(pixels.size());
339
340 mSPI.waitFully();
341 mSPI.release();
342 }

344 inline void showPixelsGammaBitShift(PixelController<RGB_ORDER> & pixels) {
345 mSPI.select();
346 startBoundary();
347 while (pixels.has(1)) {
348 // Load raw uncorrected r,g,b values.
349 uint8_t brightness, c0, c1, c2; // c0-c2 is the RGB data re-ordered for pixel
350 pixels.loadAndScale_APA102_HD(&c0, &c1, &c2, &brightness);
351 writeLed(brightness, c0, c1, c2);
352 pixels.stepDithering();
353 pixels.advanceData();
354 }
355 endBoundary(pixels.size());
356 mSPI.waitFully();
357 mSPI.release();
358 }
