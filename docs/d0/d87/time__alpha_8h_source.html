17inline float time_alphaf(u32 now, u32 start, u32 end) {
18 if (now < start) {
19 return 0.0f;
20 }
21 u32 elapsed = now - start;
22 u32 total = end - start;
23 float out = static_cast<float>(elapsed) / static_cast<float>(total);
24 return out;
25}
