16// Optimized for 2^n grid sizes in terms of both memory and performance.
17// If you are somehow running this on AVR then you probably want this if
18// you can make your grid size a power of 2.
19#define FASTLED_SCALE_UP_ALWAYS_POWER_OF_2 0 // 0 for always power of 2.
20// Uses more memory than FASTLED_SCALE_UP_ALWAYS_POWER_OF_2 but can handle
21// arbitrary grid sizes.
22#define FASTLED_SCALE_UP_HIGH_PRECISION 1 // 1 for always choose high precision.
23// Uses the most executable memory because both low and high precision versions
24// are compiled in. If the grid size is a power of 2 then the faster version is
25// used. Note that the floating point version has to be directly specified
26// because in testing it offered no benefits over the integer versions.
27#define FASTLED_SCALE_UP_DECIDE_AT_RUNTIME 2 // 2 for runtime decision.
28
29#define FASTLED_SCALE_UP_FORCE_FLOATING_POINT 3 // Warning, this is slow.

44void ScaleUp::draw(DrawContext context) {
45 if (mSurface.empty()) {
46 mSurface.resize(mDelegate->getNumLeds());
47 }

49 delegateContext.leds = mSurface.data();

53 uint16_t in_h = mDelegate->getHeight();
54 uint16_t out_w = getWidth();
55 uint16_t out_h = getHeight();
56 ;
57 if (in_w == out_w && in_h == out_h) {
58 noExpand(mSurface.data(), context.leds, in_w, in_h);
59 } else {

64void ScaleUp::expand(const CRGB *input, CRGB *output, uint16_t width,
65 uint16_t height, XYMap mXyMap) {
66#if FASTLED_SCALE_UP == FASTLED_SCALE_UP_ALWAYS_POWER_OF_2
67 fl::upscalePowerOf2(input, output, static_cast<uint8_t>(width), static_cast<uint8_t>(height), mXyMap);
68#elif FASTLED_SCALE_UP == FASTLED_SCALE_UP_HIGH_PRECISION
69 fl::upscaleArbitrary(input, output, width, height, mXyMap);
70#elif FASTLED_SCALE_UP == FASTLED_SCALE_UP_DECIDE_AT_RUNTIME
71 fl::upscale(input, output, width, height, mXyMap);
72#elif FASTLED_SCALE_UP == FASTLED_SCALE_UP_FORCE_FLOATING_POINT
73 fl::upscaleFloat(input, output, static_cast<uint8_t>(width), static_cast<uint8_t>(height), mXyMap);
74#else
75#error "Invalid FASTLED_SCALE_UP"
76#endif
77}

79void ScaleUp::noExpand(const CRGB *input, CRGB *output, uint16_t width,
80 uint16_t height) {
81 uint16_t n = mXyMap.getTotal();
82 for (uint16_t w = 0; w < width; w++) {
83 for (uint16_t h = 0; h < height; h++) {
84 uint16_t idx = mXyMap.mapToIndex(w, h);
85 if (idx < n) {
86 output[idx] = input[w * height + h];
87 }
88 }
89 }
90}
