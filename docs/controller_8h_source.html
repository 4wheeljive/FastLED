253 static CRGB computeAdjustment(uint8_t scale, const CRGB & colorCorrection, const CRGB & colorTemperature) {
254 #if defined(NO_CORRECTION) && (NO_CORRECTION==1)
255 return CRGB(scale,scale,scale);
256 #else
257 CRGB adj(0,0,0);
258
259 if(scale > 0) {
260 for(uint8_t i = 0; i < 3; ++i) {
261 uint8_t cc = colorCorrection.raw[i];
262 uint8_t ct = colorTemperature.raw[i];
263 if(cc > 0 && ct > 0) {
264 uint32_t work = (((uint32_t)cc)+1) * (((uint32_t)ct)+1) * scale;
265 work /= 0x10000L;
266 adj.raw[i] = work & 0xFF;
267 }
268 }
269 }
270
271 return adj;
272 #endif

408 void init_binary_dithering() {
409#if !defined(NO_DITHERING) || (NO_DITHERING != 1)
410 // R is the digther signal 'counter'.
411 static uint8_t R = 0;
412 ++R;
413
414 // R is wrapped around at 2^ditherBits,
415 // so if ditherBits is 2, R will cycle through (0,1,2,3)
416 uint8_t ditherBits = VIRTUAL_BITS;
417 R &= (0x01 << ditherBits) - 1;
418
419 // Q is the "unscaled dither signal" itself.
420 // It's initialized to the reversed bits of R.
421 // If 'ditherBits' is 2, Q here will cycle through (0,128,64,192)
422 uint8_t Q = 0;
423
424 // Reverse bits in a byte
425 {
426 if(R & 0x01) { Q |= 0x80; }
427 if(R & 0x02) { Q |= 0x40; }
428 if(R & 0x04) { Q |= 0x20; }
429 if(R & 0x08) { Q |= 0x10; }
430 if(R & 0x10) { Q |= 0x08; }
431 if(R & 0x20) { Q |= 0x04; }
432 if(R & 0x40) { Q |= 0x02; }
433 if(R & 0x80) { Q |= 0x01; }
434 }
435
436 // Now we adjust Q to fall in the center of each range,
437 // instead of at the start of the range.
438 // If ditherBits is 2, Q will be (0, 128, 64, 192) at first,
439 // and this adjustment makes it (31, 159, 95, 223).
440 if( ditherBits < 8) {
441 Q += 0x01 << (7 - ditherBits);
442 }
443
444 // D and E form the "scaled dither signal"
445 // which is added to pixel values to affect the
446 // actual dithering.
447
448 // Setup the initial D and E values
449 for(int i = 0; i < 3; ++i) {
450 uint8_t s = mScale.raw[i];
451 e[i] = s ? (256/s) + 1 : 0;
452 d[i] = scale8(Q, e[i]);
453#if (FASTLED_SCALE8_FIXED == 1)
454 if(d[i]) (--d[i]);
455#endif
456 if(e[i]) --e[i];
457 }
458#endif
459 }
