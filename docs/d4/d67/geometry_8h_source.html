332 vec2<T> *out_projected = nullptr) const {
333 return distance_to_line_with_point(p, start, end, out_projected);
334 }
335
336 private:
337 // Computes the closest distance from `p` to the line through `a` and `b`,
338 // and writes the projected point.
339 static float distance_to_line_with_point(vec2<T> p, vec2<T> a, vec2<T> b,
340 vec2<T> *out_projected) {
341 vec2<T> maybe;
342 vec2<T> &out_proj = out_projected ? *out_projected : maybe;
343 float dx = b.x - a.x;
344 float dy = b.y - a.y;
345 float len_sq = dx * dx + dy * dy;
346
347#pragma GCC diagnostic push
348#pragma GCC diagnostic ignored "-Wfloat-equal"
349 const bool is_zero = (len_sq == 0.0f);
350#pragma GCC diagnostic pop
351
352 if (is_zero) {
353 // a == b, the segment is a point
354 out_proj = a;
355 dx = p.x - a.x;
356 dy = p.y - a.y;
357 return sqrt(dx * dx + dy * dy);
358 }
359
360 // Project point p onto the line segment, computing parameter t
361 float t = ((p.x - a.x) * dx + (p.y - a.y) * dy) / len_sq;
362
363 // Clamp t to [0,1] to stay within the segment
364 if (t < 0.0f)
365 t = 0.0f;
366 else if (t > 1.0f)
367 t = 1.0f;
368
369 // Find the closest point
370 out_proj.x = a.x + t * dx;
371 out_proj.y = a.y + t * dy;
372
373 dx = p.x - out_proj.x;
374 dy = p.y - out_proj.y;
375 return sqrt(dx * dx + dy * dy);
376 }
